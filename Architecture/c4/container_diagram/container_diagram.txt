!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

title Container Diagram, LeNotLampe

Person(lmpusr, "LeLampe User", "A user of LeLampe, in the same room as it")

System_Boundary(lmp, "LeLampe", "Friendly intelligent robotic lamp"){
 
 Container(in,"Input Hardware","Boop Button, Potentiometer, microphone")
 Container(cv,"CV/Perception/Understanding","cv model and others that Perceive the real world, user emotion, etc. and determines proper action to take, emotion to emulate, etc.")
 Container(bev,"Physical Behavior Engine","translates textual command to matching, safe movement")
 Container(data,"Database","holds Data taken from conversations, perception")
 Container(llm,"LLM/Conversation","generates textual/audible replies to user")
 Container(late,"Latency Tracker","keeps track of end-to-end latency")
 Container(out, "Output Hardware","speakers, motor, screen, motorcontroller")
}

System_Ext(room,"Objects, Room", "room of user and everything in it")

System(met,"Metrics", "collects and analyzes Lelamp's Metrics")

Rel(in,cv,"takes input data from")
Rel(cv,data,"read, write")
Rel(llm,data,"read, write")

Rel(lmpusr,in,"interacts with")
Rel(cv,lmpusr,"observes, listens to")

Rel(cv,bev,"gives data on correct action / behaviour to take")
Rel(bev,cv,"returns request to re-determine behaviour/action if previous is illegal")

Rel(bev,out,"sends safe commands to motorcontrollers")

Rel(cv,room, "observes")

Rel(lmpusr,room,"interacts with")

Rel(met,late,"observes latency")

Rel(out,lmpusr,"responds to, gives data, interacts with")

Rel(cv,llm,"sends observed context data")

Rel(llm,out,"feeds response to speaker")

@enduml
